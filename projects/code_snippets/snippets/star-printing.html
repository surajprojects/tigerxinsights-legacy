    #include <stdio.h>

        int triangular(int number)
        {
            printf("\n");
            for (int i = 1; i < (number + 1); i++)
            {
                for (int j = 0; j < i; j++)
                {
                    printf("%c", '*');
                }
                printf("\n");
            }
        }
        
        int reversed_triangular(int number)
        {
            printf("\n");
            for (int i = 1; i < (number + 1); i++)
            {
                for (int j = (number + 1); j > i; j--)
                {
                    printf("%c", '*');
                }
                printf("\n");
            }
        }
        
        int main()
        {
            int option, lines;

                scanf("%d", &option);
        
                if (option == 1)
                {

                    triangular(lines);
                }
                else if (option == 2)
                {
                    printf("Enter the number of lines to print: ");
                    scanf("%d", &lines);
                    reversed_triangular(lines);

                else
